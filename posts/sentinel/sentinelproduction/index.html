Sentinel з”ҹдә§йғЁзҪІ#

рҹ“– еүҚзҪ®йҳ…иҜ»пјҡжң¬ж–ҮеҒҮи®ҫиҜ»иҖ…е·ІжҺҢжҸЎ Sentinel жөҒжҺ§е’ҢзҶ”ж–ӯи§„еҲҷгҖӮеҰӮжһңиҝҳдёҚзҶҹжӮүпјҢе»әи®®е…Ҳйҳ…иҜ»еүҚдёүзҜҮпјҡж ёеҝғжҰӮеҝөгҖҒжөҒжҺ§и§„еҲҷгҖҒзҶ”ж–ӯйҷҚзә§гҖӮ

дёҖгҖҒвҡЎ жөҒжҺ§е’ҢзҶ”ж–ӯйғҪй…ҚдәҶвҖ”вҖ”дҪҶж•ҙдёӘжңәеҷЁзҡ„ CPU йЈҷеҲ° 95% дәҶ#

жөҒжҺ§и§„еҲҷдҝқжҠӨзҡ„жҳҜеҚ•дёӘжҺҘеҸЈвҖ”вҖ”“getUser жҜҸз§’жңҖеӨҡ 100 дёӘ”гҖӮ“зҶ”ж–ӯи§„еҲҷдҝқжҠӨзҡ„жҳҜжҺҘеҸЈиҮӘиә«ж•…йҡңвҖ”вҖ”“getUser 50% ж…ўи°ғз”Ёе°ұзҶ”ж–ӯ”гҖӮ

дҪҶиҝҷдәӣи§„еҲҷдёҚдҝқжҠӨж•ҙдёӘжңәеҷЁвҖ”вҖ”еҰӮжһң 20 дёӘжҺҘеҸЈеҗ„иҮӘйғҪжІЎи¶…иҝҮиҮӘе·ұзҡ„ QPS йҳҲеҖјпјҢдҪҶеҠ иө·жқҘжҠҠжңәеҷЁзҡ„ CPU жү“ж»ЎдәҶвҖ”вҖ”жүҖжңүжҺҘеҸЈдёҚеҸҜз”ЁгҖӮ

зі»з»ҹи§„еҲҷпјҲSystem Ruleпјүи§ЈеҶізҡ„е°ұжҳҜиҝҷдёӘй—®йўҳвҖ”вҖ”д»Һж•ҙдёӘеә”з”Ёзҡ„еұӮйқўеҒҡиҮӘйҖӮеә”дҝқжҠӨгҖӮ

дәҢгҖҒрҹ§¬ зі»з»ҹиҮӘйҖӮеә”дҝқжҠӨвҖ”вҖ”дёҚй…Қе…·дҪ“ QPSпјҢй…Қ"еҒҘеә·жҢҮж Ү”#

2.1 дә”з§Қзі»з»ҹи§„еҲҷ#

// зі»з»ҹи§„еҲҷвҖ”вҖ”еҜ№ж•ҙдёӘжңҚеҠЎз”ҹж•ҲвҖ”вҖ”дёҚжҳҜй’ҲеҜ№жҹҗдёӘиө„жәҗ
SystemRule systemRule = new SystemRule();

// в‘  Load дҝқжҠӨвҖ”вҖ”зі»з»ҹиҙҹиҪҪпјҲд»… Linuxпјүи¶…иҝҮйҳҲеҖјж—¶йҷҗжөҒ
systemRule.setHighestSystemLoad(4.0);      // CPU ж ёж•°вҖ”вҖ”еҰӮ 4 ж ё CPU
// зі»з»ҹ Load > 4.0 ж—¶вҖ”вҖ”жүҖжңүе…ҘеҸЈ QPS иҮӘеҠЁйҷҚеҲ° (Load / еҪ“еүҚLoad) * еҪ“еүҚQPS

// в‘Ў CPU дҪҝз”ЁзҺҮдҝқжҠӨ
systemRule.setHighestCpuUsage(0.8);        // CPU дҪҝз”ЁзҺҮ > 80% ж—¶вҖ”вҖ”жӢ’з»қж–°зҡ„е…ҘеҸЈиҜ·жұӮ

// в‘ў е№іеқҮ RT дҝқжҠӨ
systemRule.setAvgRt(100);                  // жүҖжңүе…ҘеҸЈзҡ„е№іеқҮ RT > 100ms ж—¶вҖ”вҖ”йҷҗжөҒ

// в‘Ј жңҖеӨ§е№¶еҸ‘зәҝзЁӢж•°
systemRule.setMaxThread(200);              // е№¶еҸ‘зәҝзЁӢж•° > 200вҖ”вҖ”жӢ’з»қж–°иҜ·жұӮ

// в‘Ө е…ҘеҸЈ QPSвҖ”вҖ”иҝҷдёӘжңҖзӣҙжҺҘ
systemRule.setQps(500);                    // жүҖжңүе…ҘеҸЈпјҲдёҚз®ЎжҳҜе“ӘдёӘиө„жәҗпјүвҖ”вҖ”жҖ» QPS > 500
зі»з»ҹи§„еҲҷжҢҮж ҮйҳҲеҖје»әи®®йҖӮз”ЁеңәжҷҜ
Loadзі»з»ҹ LoadвүӨ CPU ж ёж•°Linux зҺҜеўғвҖ”вҖ”жңҖжҺЁиҚҗ
CPU дҪҝз”ЁзҺҮCPU usageвүӨ 80%и·Ёе№іеҸ°вҖ”вҖ”е’Ң Load дәҢйҖүдёҖ
е№іеқҮ RTжүҖжңүе…ҘеҸЈе№іеқҮ RTвүӨ жӯЈеёёеҖј Г— 2жңҚеҠЎеҸҳж…ўж—¶иҮӘеҠЁйҷҚ QPS
е№¶еҸ‘зәҝзЁӢж•°е№¶еҸ‘зәҝзЁӢж•°вүӨ зәҝзЁӢжұ еӨ§е°ҸйҳІжӯўзәҝзЁӢжұ ж»Ў
е…ҘеҸЈ QPSжҖ» QPSвүӨ еҺӢжөӢеҖј Г— 80%з®ҖеҚ•зІ—жҡҙвҖ”вҖ”е…ңеә•ж–№жЎҲ

2.2 зі»з»ҹи§„еҲҷзҡ„жңҖдҪіз»„еҗҲ#

@Component
public class SystemRuleInitializer implements ApplicationRunner {

    @Override
    public void run(ApplicationArguments args) {
        List<SystemRule> rules = new ArrayList<>();

        // и§„еҲҷ 1пјҡLoad дҝқжҠӨвҖ”вҖ”зі»з»ҹиҙҹиҪҪиҝҮй«ҳж—¶иҮӘеҠЁйҷҚ QPS
        SystemRule loadRule = new SystemRule();
        loadRule.setHighestSystemLoad(4.0);
        rules.add(loadRule);

        // и§„еҲҷ 2пјҡе№іеқҮ RT дҝқжҠӨвҖ”вҖ”жҺҘеҸЈеҸҳж…ўж—¶иҮӘеҠЁеҮҸйҖҹ
        SystemRule rtRule = new SystemRule();
        rtRule.setAvgRt(200);
        rules.add(rtRule);

        // и§„еҲҷ 3пјҡе№¶еҸ‘зәҝзЁӢж•°дҝқжҠӨвҖ”вҖ”йҳІжӯўзәҝзЁӢжұ ж»Ў
        SystemRule threadRule = new SystemRule();
        threadRule.setMaxThread(300);
        rules.add(threadRule);

        SystemRuleManager.loadRules(rules);
    }
}

зі»з»ҹи§„еҲҷжҳҜж•ҙдёӘ JVM зә§еҲ«зҡ„вҖ”вҖ”дёҚйңҖиҰҒжҢҮе®ҡиө„жәҗеҗҚгҖӮе®ғзҡ„дҪңз”ЁиҢғеӣҙжҳҜжүҖжңүе…ҘеҸЈпјҲжүҖжңүз»ҸиҝҮ Sentinel дҝқжҠӨзҡ„е…ҘеҸЈжөҒйҮҸзҡ„жұҮжҖ»пјүгҖӮ

2.3 зі»з»ҹи§„еҲҷе’ҢжөҒжҺ§и§„еҲҷзҡ„е…ізі»вҖ”вҖ”и°Ғе…Ҳз”ҹж•Ҳпјҹ#

иҜ·жұӮиҝӣжқҘ вҶ’
  в‘  е…ҲжЈҖжҹҘзі»з»ҹи§„еҲҷвҖ”вҖ”ж•ҙдёӘжңәеҷЁ CPU 80% дәҶеҗ—пјҹ
    вҶ’ еҰӮжһң CPU 80%пјҡзӣҙжҺҘжӢ’з»қпјҲдёҚз®ЎдҪ иҝҷдёӘжҺҘеҸЈиҮӘе·ұзҡ„ QPS еӨҡе°‘пјү
    вҶ’ еҰӮжһң CPU жӯЈеёёпјҡз»§з»ӯ
  в‘Ў еҶҚжЈҖжҹҘжөҒжҺ§и§„еҲҷвҖ”вҖ”иҝҷдёӘжҺҘеҸЈзҡ„ QPS иҝҮ 100 дәҶеҗ—пјҹ
    вҶ’ еҰӮжһңи¶…дәҶпјҡжӢ’з»қ
    вҶ’ еҰӮжһңжІЎи¶…пјҡйҖҡиҝҮ

зі»з»ҹи§„еҲҷдјҳе…Ҳзә§жңҖй«ҳвҖ”вҖ”е®ғжҳҜе…ЁеұҖе…ңеә•гҖӮеҚ•дёӘжҺҘеҸЈиў«йҷҗжөҒдәҶеҸӘжҳҜдёҖдёӘжҺҘеҸЈдёҚеҸҜз”ЁвҖ”вҖ”дҪҶжңәеҷЁжү“жҢӮдәҶжҳҜжүҖжңүжҺҘеҸЈдёҚеҸҜз”ЁгҖӮ

дёүгҖҒрҹ”Ҙ зғӯзӮ№еҸӮж•°йҷҗжөҒвҖ”вҖ”д»Җд№Ҳж—¶еҖҷйңҖиҰҒ"жҢүеҸӮж•°иҝҮж»Ө"пјҹ#

3.1 дёәд»Җд№ҲйңҖиҰҒеҸӮж•°зә§йҷҗжөҒпјҹ#

QPS йҷҗжөҒжҳҜ"иҝҷдёӘжҺҘеҸЈжҖ»е…ұжҜҸз§’жңҖеӨҡ 1000 дёӘ"гҖӮдҪҶеҪ“ 1000 дёӘиҜ·жұӮдёӯ 900 дёӘйғҪеңЁжҹҘеҗҢдёҖдёӘзғӯзӮ№е•Ҷе“ҒпјҲID=10001пјүпјҢиҝҷдёӘе•Ҷе“ҒжүҖеңЁзҡ„ж•°жҚ®еә“еҲҶзүҮеҸҜиғҪиў«жү“зҲҶгҖӮ

зғӯзӮ№еҸӮж•°йҷҗжөҒи®©дҪ еҜ№зғӯй—ЁеҸӮж•°еҖјеҚ•зӢ¬и®ҫйҳҲеҖјпјҡ

4.3 иҮӘе®ҡд№ү Gateway йҷҗжөҒеҲҶз»„вҖ”вҖ”жҢүи·Ҝеҫ„жҲ–жҢү IP#

@Configuration
public class GatewaySentinelConfig {

    @PostConstruct
    public void initGatewayRules() {
        Set<GatewayFlowRule> rules = new HashSet<>();

        // и§„еҲҷ 1пјҡжҢүи·Ҝз”ұйҷҗжөҒвҖ”вҖ”user-service и·Ҝз”ұж•ҙдҪ“ QPS дёҚи¶…иҝҮ 1000
        GatewayFlowRule userServiceRule = new GatewayFlowRule("user-service");
        userServiceRule.setCount(1000);
        userServiceRule.setIntervalSec(1);
        rules.add(userServiceRule);

        // и§„еҲҷ 2пјҡжҢү API еҲҶз»„вҖ”вҖ”жҠҠеӨҡдёӘи·Ҝеҫ„жҳ е°„еҲ°дёҖдёӘз»„з»ҹдёҖйҷҗжөҒ
        ApiDefinition apiDef = new ApiDefinition("user-read-api")
                .setPredicateItems(new HashSet<>(Arrays.asList(
                        new ApiPathPredicateItem()
                                .setPattern("/api/users/**")
                                .setMatchStrategy(SentinelGatewayConstants.URL_MATCH_STRATEGY_PREFIX)
                )));
        GatewayApiDefinitionManager.loadApiDefinitions(Collections.singleton(apiDef));

        GatewayFlowRule apiRule = new GatewayFlowRule("user-read-api");
        apiRule.setCount(500);           // иҝҷз»„ API жҖ»е…ұ 500 QPS
        apiRule.setIntervalSec(1);
        apiRule.setBurst(2);             // еҸӮж•°зҙўеј•пјҲburst жЁЎејҸпјү
        rules.add(apiRule);

        // и§„еҲҷ 3пјҡжҢү IP йҷҗжөҒвҖ”вҖ”жҜҸдёӘ IP жңҖеӨҡ 10 QPS
        GatewayFlowRule ipRule = new GatewayFlowRule("user-service");
        ipRule.setCount(10);
        ipRule.setIntervalSec(1);
        ipRule.setParamItem(new GatewayParamFlowItem()
                .setParseStrategy(SentinelGatewayConstants.PARAM_PARSE_STRATEGY_CLIENT_IP));
        rules.add(ipRule);

        GatewayRuleManager.loadRules(rules);
    }
}

рҹҺҜ жҖ»з»“#

  1. зі»з»ҹи§„еҲҷжҳҜжңҖеҗҺзҡ„йҳІзәҝпјҡжҺҘеҸЈйҷҗжөҒе’ҢзҶ”ж–ӯдҝқжҠӨеҚ•дёӘиө„жәҗвҖ”вҖ”зі»з»ҹи§„еҲҷдҝқжҠӨж•ҙдёӘ JVMгҖӮLoad > CPU ж ёж•°жҲ– CPU дҪҝз”ЁзҺҮ > 80% ж—¶вҖ”вҖ”жүҖжңүе…ҘеҸЈз»ҹдёҖйҷҗжөҒгҖӮзі»з»ҹи§„еҲҷдјҳе…Ҳзә§жңҖй«ҳвҖ”вҖ”е…ЁеұҖе…ңеә•гҖӮ

  2. и§„еҲҷжҢҒд№…еҢ–жҳҜеҝ…йЎ»зҡ„вҖ”вҖ”еҲ«з”Ёй»ҳи®ӨеҶ…еӯҳжЁЎејҸпјҡSentinel й»ҳи®Өи§„еҲҷеӯҳеңЁеҶ…еӯҳвҖ”вҖ”жңҚеҠЎйҮҚеҗҜеҗҺдёўе…үгҖӮз”Ё Nacos Push жЁЎејҸвҖ”вҖ”Dashboard ж”№и§„еҲҷ вҶ’ еҶҷ Nacos вҶ’ жҺЁжүҖжңүе®һдҫӢвҖ”вҖ”йҮҚеҗҜд№ҹдёҚдёўгҖӮ

  3. Gateway йӣҶжҲҗ Sentinel еҒҡзҪ‘е…іеұӮйҷҗжөҒпјҡGateway Filter дёӯеҶ…зҪ®дәҶ SentinelвҖ”вҖ”жҢүи·Ҝз”ұ/API еҲҶз»„/IP дёүз§Қз»ҙеәҰйҷҗжөҒгҖӮе’Ңеҫ®жңҚеҠЎеұӮзҡ„ Sentinel дёҚеҶІзӘҒвҖ”вҖ”еҸҢдҝқйҷ©гҖӮ

  4. зғӯзӮ№еҸӮж•°йҷҗжөҒжҳҜзІҫз»ҶеҢ–е·Ҙе…·пјҡзғӯй—Ё productId ж”ҫжӣҙе®Ҫзҡ„йҳҲеҖјвҖ”вҖ”зҲ¬иҷ«зӣҜдёҠзҡ„ ID ж”ҫжӣҙдёҘзҡ„йҳҲеҖјгҖӮеҲ«з»ҷжүҖжңүеҸӮж•°еҖјдёҖи§ҶеҗҢд»ҒгҖӮ
